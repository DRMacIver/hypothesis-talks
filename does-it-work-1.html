What is a sorted list?

  • [1, 2] ✓
  • [2, 1] ✘
  • [1, 2, 3] ✓
  • [3, 2, 1] ✘
  • [1, 2, 2] ✓
  • [2, 1, 2] ✘

How do we sort a list?

  • [1, 2] → [1, 2]
  • [2, 1] → [1, 2]
  • [1, 2, 3] → [1, 2, 3]
  • [3, 2, 1] → [1, 2, 3]
  • [1, 2, 2] → [1, 2, 2]
  • [2, 1, 2] → [1, 2, 2]

def bubble_sort(ls):
    for i in range(1, len(ls)):
        if ls[i - 1] > ls[i]:
            ls[i - 1], ls[i] = ls[i], ls[i - 1]


def sort(ls):
    result = list(ls)
    bubble_sort(result)
    return result

def is_sorted(ls):
    return all(
        ls[i - 1] <= ls[i]
        for i in range(1, len(ls))
    )

def merge_sorted_lists(x, y):
    result = []
    i = 0
    j = 0
    while i < len(x) and j < len(y):
        if x[i] <= y[j]:
            result.append(x[i])
            i += 1
        else:
            result.append(y[j])
            j += 1
    return result
